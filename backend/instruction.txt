# vào backend
cd backend

# Khởi tạo csdl 
python init_db.py

# chạy app 
python main.py

# push lên docker hub
docker build -t hoangtranthanh/qldapm-backend:latest .
docker push hoangtranthanh/qldapm-backend:latest


# test
python -m unittest discover -s tests -p "*.py" -v

# deploy start command python3 render
bash -c "python3 init_db.py && python3 main.py"

# format lại code
black --check app tests || exit 1

# Lint code check lỗi syntax nghiêm trọng
flake8 || exit 1

# Scan security
bandit -r ./app --severity-level high

# Scan dependency vulnerabilities
safety check -r requirements.txt
